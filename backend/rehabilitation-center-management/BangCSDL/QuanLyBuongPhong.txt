-- CƠ SỞ DỮ LIỆU QUẢN LÝ BUỒNG, PHÒNG

-- 1. Thông tin buồng phòng
CREATE TABLE BuongPhong (
    MaPhong VARCHAR2(50) PRIMARY KEY,
    TenPhong VARCHAR2(255),
    LoaiPhong VARCHAR2(100),
    SoLuong VARCHAR2(10),
    SucChua VARCHAR2(10),
    TinhTrang VARCHAR2(100),
    NguoiPhuTrach VARCHAR2(255),
    SoLuongHocVienDangDung VARCHAR2(10)
);

-- 2. Lịch sử lưu trú
CREATE TABLE LichSuLuuTru (
    ID NUMBER GENERATED BY DEFAULT ON NULL AS IDENTITY PRIMARY KEY,
    MaHocVien VARCHAR2(50),
    TenHocVien VARCHAR2(255),
    ThoiGianVaoPhong DATE,
    ThoiGianRaPhong DATE,
    GhiChu VARCHAR2(255),
    FOREIGN KEY (MaHocVien) REFERENCES BuongPhong(MaPhong)
);
